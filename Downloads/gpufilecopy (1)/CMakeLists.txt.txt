cmake_minimum_required(VERSION 3.20)

project(filecopy LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CUDA_STANDARD 17)

find_package(MPI REQUIRED)

add_executable(filecopy_cuda_mpi main.cu)

target_link_libraries(filecopy_cuda_mpi
    PRIVATE
    MPI::MPI_CXX
)

set_target_properties(filecopy_cuda_mpi PROPERTIES
    CUDA_SEPARABLE_COMPILATION ON
)

if (MSVC)
    target_link_options(filecopy_cuda_mpi PRIVATE /INCREMENTAL:NO)
endif()
